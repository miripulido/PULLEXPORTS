  muscle: ["camaro","mustang","challenger","charger","corvette"],
  pickup: ["ram","f-150","f-250","f150","f250","silverado","sierra","tundra","tacoma","ranger","bronco sport"],

function getCategory(car) {
  const t = (car.title || car.model || "").toLowerCase();
  for (const [cat, keywords] of Object.entries(CAT_MAP)) {
    if (keywords.some(k => t.includes(k))) return cat;
  }
  return "otro";
}

// ── Carga de datos ─────────────────────────────────────────────────
async function loadData() {
  document.getElementById("grid").innerHTML = '<div class="loader"><span></span><span></span><span></span></div>';

  // Intenta Supabase primero, luego JSON local
  if (CONFIG.supabase_url && CONFIG.supabase_key) {
    await loadFromSupabase();
  } else {
    await loadFromJSON();
  }
}

async function loadFromSupabase() {
  try {
    const url = `${CONFIG.supabase_url}/rest/v1/cars?select=*&order=opportunity_score.desc&limit=200`;
    const res = await fetch(url, {
      headers: {
        "apikey": CONFIG.supabase_key,
        "Authorization": `Bearer ${CONFIG.supabase_key}`,
      }
    });
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    const data = await res.json();
    ALL_CARS = data.map(enrichCar);
    applyFilters();
  } catch(e) {
    console.error("Supabase error:", e);
    await loadFromJSON();
  }
}

async function loadFromJSON() {
  try {
    const res = await fetch(CONFIG.local_json + "?t=" + Date.now());
    if (!res.ok) throw new Error("No se encontró cars_data.json");
    const data = await res.json();
    ALL_CARS = (data.cars || []).map(enrichCar);
    applyFilters();
  } catch(e) {
    // Si no hay datos, mostrar demo
    loadDemoData();
  }
}

function enrichCar(c) {
  const totalCost = (c.price||0) + COST_IMPORT;
  const spain = c.spain_ref_price || 0;
  const profit = spain > 0 ? spain - totalCost : (c.estimated_profit || 0);
  const marginPct = spain > 0 ? Math.round((profit / spain) * 100) : (c.margin_pct || 0);
  return {
    ...c,
    category: getCategory(c),
    total_cost: totalCost,
    profit: profit,
    margin_pct: marginPct,
    opportunity_score: c.opportunity_score || 50,
  };
}

    return true;
  });

  renderGrid();

      <div class="badge-src">${c.source||""}</div>
      <div class="badge-score ${scoreClass(sc)}">★ ${sc}</div>
    </div>
    <div class="card-body">
      <div class="card-title">${c.title || c.make + " " + c.model}</div>
      <div class="card-sub">${c.year||"—"} · ${c.cv ? c.cv + " CV" : "—"} · ${c.km ? (c.km/1000).toFixed(0)+"k km" : "—"}</div>
      <div class="tags">
        <span class="tag">${flag} ${c.country}</span>
        ${c.category !== "otro" ? `<span class="tag hl">${c.category}</span>` : ""}
      </div>
      <div class="price-row">
        <div>
          <div class="buy-price">${fmt(c.price)}</div>
          <div style="font-size:11px;color:var(--muted);margin-top:2px">+ ${fmt(COST_IMPORT)} gastos</div>
        </div>
        <div class="profit-info">
          <div class="mchip ${marginClass(margin)}">+${margin}%</div>
          ${profit > 0 ? `<div class="profit-num">Benef. est.: <strong>+${fmt(profit)}</strong></div>` : ""}
        </div>
      </div>
    </div>
  </div>`;
}

function closeModal(e) {

  }

// ── Stats ──────────────────────────────────────────────────────────
function updateStats() {
  document.getElementById("s-total").textContent = FILTERED.length;
  if (!FILTERED.length) return;
  const avgM = Math.round(FILTERED.reduce((a,c) => a + (c.margin_pct||0), 0) / FILTERED.length);
  document.getElementById("s-margin").textContent = "+" + avgM + "%";
  const best = FILTERED.reduce((a,b) => (a.profit||0) > (b.profit||0) ? a : b);
  document.getElementById("s-best").textContent = "+" + fmt(best.profit||0);
  document.getElementById("s-bestname").textContent = (best.make||"") + " " + (best.model||"");
  const now = new Date();
  document.getElementById("s-time").textContent = now.toLocaleTimeString("es-ES", {hour:"2-digit",minute:"2-digit"});
}

function updateCounts() {
  ["DE","FR","BE","PL","PT"].forEach(c => {
    const el = document.getElementById("cnt-"+c);
    if (el) el.textContent = ALL_CARS.filter(x => x.country === c).length;
  });
}

// ── Export CSV ─────────────────────────────────────────────────────
function exportCSV() {
  const cols = ["title","year","km","price","country","cv","opportunity_score","profit","spain_ref_price","url","source"];
  const rows = [cols.join(";")];
  FILTERED.forEach(c => rows.push(cols.map(k => c[k]||"").join(";")));
  const blob = new Blob([rows.join("\n")], {type:"text/csv;charset=utf-8;"});
  const a = document.createElement("a");
  a.href = URL.createObjectURL(blob);
  a.download = `autoexport_${new Date().toISOString().split("T")[0]}.csv`;
  a.click();
}
